async function analisis(){ if(historyMusic.length === 0){ alert("Masukkan riwayat lagu terlebih dahulu"); return; } const loading = document.getElementById("loading"); loading.style.display = "block"; document.getElementById("result").innerText = "Mencari rekomendasi lagu..."; const daftarLagu = historyMusic .map((d,i)=> (i+1) + ". " + d.song + " - " + d.artist ) .join("\n"); try{ const res = await fetch( "https://project-lagu-backend-indahh12s-projects.vercel.app/api/analyze", { method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ songs: daftarLagu }) } ); console.log("Status:", res.status); const json = await res.json(); console.log("Response:", json); if(json.result){ hasilAnalisis = json.result; document.getElementById("result").innerText = json.result; document.getElementById("generateBtn") .style.display = "inline-block"; }else{ document.getElementById("result").innerText = json.error || "AI gagal menganalisis"; } }catch(err){ console.error("ERROR:", err); alert("Error: " + err.message); document.getElementById("result").innerText = err.message; }finally{ loading.style.display = "none"; } }
